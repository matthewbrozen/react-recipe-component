import React from 'react';
import PropTypes from 'prop-types';

/**
 * Represents the current score of a quiz.
 */
class Score extends React.Component {

    /**
     * Render the score out of the current possible maximum score
     * @returns {JSX}
     */
    renderCurrentMax() {
        if ((typeof this.props.score !== 'number') || (typeof this.props.currentMax !== 'number')) {
            return null;
        }
        return (
            <div className='quiz-score__values quiz-score__values--current-max'>
                <span className='quiz-score__field'>Score</span>
                <span className='quiz-score__value'>{this.props.score}</span>
                <span className='quiz-score__of'>of</span>
                <span className='quiz-score__total'>{this.props.currentMax}</span>
                <span className='quiz-score__field-suffix'>possible for the questions so far</span>
            </div>
        );
    }

    /**
     * Render the number of questions correct so far.
     * @returns {JSX}
     */
    renderTotalCorrect() {
        if ((typeof this.props.totalCorrect !== 'number') || (typeof this.props.current !== 'number')) {
            return null;
        }
        const questionNum = this.props.current;
        return (
            <div className='quiz-score__values quiz-score__values--total-correct'>
                <span className='quiz-score__field'></span>
                <span className='quiz-score__value'>{this.props.totalCorrect}</span>
                <span className='quiz-score__of'>of</span>
                <span className='quiz-score__total'>{questionNum}</span>
                <span className='quiz-score__field-suffix'>correct so far</span>
            </div>
        );
    }

    /**
     * Render the question x of y
     * @returns {JSX}
     */
    renderQuestionNumber() {
        if ((typeof this.props.current !== 'number') || (typeof this.props.totalQuestions !== 'number')) {
            return null;
        }
        const questionNum = this.props.current + 1;
        return (
            <div className='quiz-score__values quiz-score__values--question'>
                <span className='quiz-score__field'>Question</span>
                <span className='quiz-score__value'>{questionNum}</span>
                <span className='quiz-score__of'>of</span>
                <span className='quiz-score__total'>{this.props.totalQuestions}</span>
            </div>
        );
    }

    /**
     * Render the score out of the maximum possible for the whole quiz
     * @returns {JSX}
     */
    renderScoreMax() {
        if ((typeof this.props.score !== 'number') || (typeof this.props.scoreMax !== 'number')) {
            return null;
        }
        return (
            <div className='quiz-score__values quiz-score__values--score-max'>
                <span className='quiz-score__field'>Score</span>
                <span className='quiz-score__value'>{this.props.score}</span>
                <span className='quiz-score__of'>of</span>
                <span className='quiz-score__total'>{this.props.scoreMax}</span>
                <span className='quiz-score__field-suffix'>possible for the full quiz</span>
            </div>
        );
    }

    /**
     * Render the score
     * @returns {JSX}
     */
    render() {
        return (
            <div className='quiz-score'>
                {this.props.children}
                {this.renderQuestionNumber()}
                {this.renderTotalCorrect()}
                {this.renderCurrentMax()}
                {this.renderScoreMax()}
            </div>
        );
    }
}

/**
 * @prop {function} children Allow passing children through.
 * @prop {number} current The zero indexed question number.
 * @prop {number} currentMax The highest score the user could have achieved by this point.
 * @prop {number} score The user's current score.
 * @prop {number} scoreMax The highest score the user can get for the whole quiz.
 * @prop {number} totalCorrect The number of questions the user has got correct so far.
 * @prop {number} totalQuestions The total number of questions.
 */
Score.propTypes = {
    children: PropTypes.func,
    current: PropTypes.number,
    currentMax: PropTypes.number,
    score: PropTypes.number,
    scoreMax: PropTypes.number,
    totalCorrect: PropTypes.number,
    totalQuestions: PropTypes.number
};

export default Score;
